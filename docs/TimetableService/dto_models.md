# Модель дисциплины
```json
{
    "time": "hh:mm-hh:mm",
    "name": "Имя дисциплины",
    "teacherName": "Преподователь",
    "groupName": "Группа",
    "audienceId": "Аудитория",
    "subgroup": 0,    // 0 - общая пара, 1,2 - по подгруппам        
    "type": "enum" 
    // lec - лекция
    // prac - практика
    // exam - экзамен
    // lab - лабораторная работа
    // hol - выходной
    // cred - зачёт
    // cons - консультация
    // fepo - федеральный экзамен профессионального образования
    // none - N/A
}
```

# Модель дня расписания для группы
```json
{
    "date": "dd.MM.yyyy",
    "groupName": "Имя группы",
    "disciplines": [
        // массив дисциплин
    ]
}
```

# Модель недели
```json
{
    "id": 1, // номер недели
    "from": "dd.MM.yyyy", // с какого дня эта неделя
    "to": "dd.MM.yyyy",  // по какой день эта неделя
    "dayNames": {
        "dd.MM.yyyy": "Понедельник",
        "dd.MM.yyyy": "Вторник",
        "dd.MM.yyyy": "Среда",
        "dd.MM.yyyy": "Четверг",
        "dd.MM.yyyy": "Пятница",
        "dd.MM.yyyy": "Суббота",
        "dd.MM.yyyy": "Воскресенье"
    } // словарь сопоставляющий каждый день недели с его названием
}
```

# Модель факультета и его групп
```json
{
    "facultyName": "Имя факльтета/института",
    "groups": [
        "first group",
        "second group"
        // ... массив имён групп, соответствующих данному факультету/институту
    ]
} 
```
